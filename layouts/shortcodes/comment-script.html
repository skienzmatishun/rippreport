package main import ( "errors" "io/ioutil" "os" ) // Define the shortcode function with a map of parameters func CommentScriptShortcode(params map[string]string) (string, error) { dir, err := os.Getwd() if err != nil { return "", err } files, err := ioutil.ReadDir(dir) if err != nil { return "", err } count := 0 for _, file := range files { if !file.IsDir() { count++ } } if count == 1 { return ` `, nil } else { return "", errors.New("There are no files in the current directory or more than one file.") } }
